Create virtual Environment
python -m venv environmentName1

Activate virtual Environment
Windows:
    .\environmentName1\Scripts\Activate.ps1
Command Prompt:
    .\environmentName1\Scripts\activate.bat
Bash or Linux:
    .\environmentName1\Scripts\activate
